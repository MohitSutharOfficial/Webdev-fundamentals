2. if Statement

The most basic conditional statement. It executes a block of code only if the specified condition is true.

Basic Syntax

// Basic if statement if (condition) { // Code to execute if condition is true } // Examples let age = 18; if (age >= 18) { console.log("You are an adult"); } let isRaining = true; if (isRaining) { console.log("Don't forget your umbrella!"); } // With variables let temperature = 25; if (temperature > 30) { console.log("It's hot outside"); } // With functions function isEven(number) { if (number % 2 === 0) { return true; } }

Truthy and Falsy Values

// Falsy values in JavaScript if (false) { /* won't execute */ } if (0) { /* won't execute */ } if (-0) { /* won't execute */ } if (0n) { /* won't execute */ } if ("") { /* won't execute */ } if ('') { /* won't execute */ } if (null) { /* won't execute */ } if (undefined) { /* won't execute */ } if (NaN) { /* won't execute */ } // Truthy values (everything else) if (true) { console.log("Executes"); } if (1) { console.log("Executes"); } if ("hello") { console.log("Executes"); } if ([]) { console.log("Executes - empty array is truthy"); } if ({}) { console.log("Executes - empty object is truthy"); } if (function() {}) { console.log("Executes - function is truthy"); } // Practical examples let username = "john_doe"; if (username) { console.log(`Welcome, ${username}!`); } let items = []; if (items.length) { console.log("You have items in your cart"); }

3. if...else Statement

Provides an alternative block of code to execute when the condition is false.

Basic if...else

// Basic if...else syntax if (condition) { // Code if condition is true } else { // Code if condition is false } // Examples let age = 16; if (age >= 18) { console.log("You can vote"); } else { console.log("You cannot vote yet"); } let score = 85; if (score >= 90) { console.log("Grade: A"); } else { console.log("Grade: Not an A"); } // With complex conditions let weather = "sunny"; if (weather === "sunny") { console.log("Great day for a picnic!"); } else { console.log("Maybe stay indoors"); } // Boolean conditions let isLoggedIn = false; if (isLoggedIn) { console.log("Welcome to your dashboard"); } else { console.log("Please log in"); }

if...else if...else Chain

// Multiple conditions let score = 85; if (score >= 90) { console.log("Grade: A"); } else if (score >= 80) { console.log("Grade: B"); } else if (score >= 70) { console.log("Grade: C"); } else if (score >= 60) { console.log("Grade: D"); } else { console.log("Grade: F"); } // Weather example let temperature = 22; if (temperature > 30) { console.log("It's hot! Stay hydrated."); } else if (temperature > 20) { console.log("Perfect weather!"); } else if (temperature > 10) { console.log("A bit chilly, wear a jacket."); } else { console.log("It's cold! Bundle up."); } // User role example let userRole = "admin"; if (userRole === "admin") { console.log("Full access granted"); } else if (userRole === "editor") { console.log("Edit access granted"); } else if (userRole === "viewer") { console.log("Read-only access"); } else { console.log("Access denied"); } // Time-based example let hour = new Date().getHours(); if (hour < 12) { console.log("Good morning!"); } else if (hour < 17) { console.log("Good afternoon!"); } else if (hour < 21) { console.log("Good evening!"); } else { console.log("Good night!"); }

4. switch Statement

An alternative to if...else chains when comparing a single value against multiple possible values.

Basic switch Syntax

// Basic switch statement switch (expression) { case value1: // Code for value1 break; case value2: // Code for value2 break; default: // Code if no cases match } // Day of week example let dayOfWeek = 3; switch (dayOfWeek) { case 1: console.log("Monday"); break; case 2: console.log("Tuesday"); break; case 3: console.log("Wednesday"); break; case 4: console.log("Thursday"); break; case 5: console.log("Friday"); break; case 6: console.log("Saturday"); break; case 7: console.log("Sunday"); break; default: console.log("Invalid day"); } // String comparison let fruit = "apple"; switch (fruit) { case "apple": console.log("Red or green fruit"); break; case "banana": console.log("Yellow fruit"); break; case "orange": console.log("Orange fruit"); break; default: console.log("Unknown fruit"); }

Advanced switch Patterns

// Fall-through behavior (no break) let grade = 'B'; switch (grade) { case 'A': case 'B': console.log("Excellent work!"); break; case 'C': case 'D': console.log("Good effort!"); break; case 'F': console.log("Keep trying!"); break; default: console.log("Invalid grade"); } // Multiple cases with same action let month = 12; switch (month) { case 12: case 1: case 2: console.log("Winter"); break; case 3: case 4: case 5: console.log("Spring"); break; case 6: case 7: case 8: console.log("Summer"); break; case 9: case 10: case 11: console.log("Fall"); break; default: console.log("Invalid month"); } // Switch with expressions let operation = '+'; let a = 10, b = 5; let result; switch (operation) { case '+': result = a + b; break; case '-': result = a - b; break; case '*': result = a * b; break; case '/': result = b !== 0 ? a / b : 'Cannot divide by zero'; break; default: result = 'Invalid operation'; } console.log(`${a} ${operation} ${b} = ${result}`); // Switch with function calls function processAction(action) { switch (action) { case 'save': return saveFile(); case 'load': return loadFile(); case 'delete': return deleteFile(); default: return 'Unknown action'; } }

‚ö†Ô∏è Important: Break Statements

Always remember to use break statements in switch cases to prevent fall-through behavior, unless intentionally desired.

5. Ternary Operator

A shorthand way to write simple if...else statements in a single line.

Basic Ternary Syntax

// Ternary operator syntax condition ? valueIfTrue : valueIfFalse; // Basic examples let age = 20; let status = age >= 18 ? "adult" : "minor"; console.log(status); // "adult" let isWeekend = true; let activity = isWeekend ? "relax" : "work"; console.log(activity); // "relax" // Inline usage let temperature = 25; console.log(temperature > 30 ? "Hot" : "Not hot"); // With function calls let score = 85; let grade = score >= 90 ? getHighGrade() : getLowGrade(); // Assigning values let price = 100; let discount = price > 50 ? 0.1 : 0.05; let finalPrice = price * (1 - discount);

Nested Ternary Operators

// Nested ternary (use sparingly) let score = 85; let grade = score >= 90 ? 'A' : score >= 80 ? 'B' : score >= 70 ? 'C' : score >= 60 ? 'D' : 'F'; // More readable with parentheses let weather = "sunny"; let activity = weather === "sunny" ? "go outside" : (weather === "rainy" ? "stay inside" : "check weather"); // Complex conditions let user = { age: 25, isPremium: true }; let accessLevel = user.age >= 18 ? (user.isPremium ? "premium" : "standard") : "restricted"; // Ternary with logical operators let isLoggedIn = true; let hasPermission = false; let canAccess = isLoggedIn && hasPermission ? "yes" : "no"; // Multiple ternary operators let hour = 14; let greeting = hour < 12 ? "Good morning" : hour < 17 ? "Good afternoon" : "Good evening" ;

When to Use Ternary vs if...else

Use Ternary When:

  • Simple conditions
  • Assigning values
  • Short expressions
  • Inline operations

Use if...else When:

  • Complex logic
  • Multiple statements
  • Better readability needed
  • Multiple conditions
// Good ternary usage let status = isOnline ? "online" : "offline"; let message = hasErrors ? "Error occurred" : "Success"; // Bad ternary usage (too complex) let result = condition1 ? (condition2 ? value1 : (condition3 ? value2 : value3)) : (condition4 ? value4 : value5); // Better as if...else let result; if (condition1) { if (condition2) { result = value1; } else if (condition3) { result = value2; } else { result = value3; } } else if (condition4) { result = value4; } else { result = value5; }

6. Logical Operators in Conditionals

Combine multiple conditions using logical operators: AND (&&), OR (||), and NOT (!).

AND Operator (&&)

// AND operator - all conditions must be true let age = 25; let hasLicense = true; if (age >= 18 && hasLicense) { console.log("You can drive"); } // Multiple AND conditions let username = "admin"; let password = "secret123"; let isActive = true; if (username === "admin" && password === "secret123" && isActive) { console.log("Access granted"); } // Short-circuit evaluation let user = { name: "John" }; if (user && user.name) { console.log(`Hello, ${user.name}`); } // Practical examples let balance = 1000; let amount = 500; let hasPermission = true; if (balance >= amount && hasPermission) { console.log("Transaction approved"); balance -= amount; } else { console.log("Transaction denied"); }

OR Operator (||)

// OR operator - at least one condition must be true let isWeekend = false; let isHoliday = true; if (isWeekend || isHoliday) { console.log("No work today!"); } // Multiple OR conditions let paymentMethod = "credit"; if (paymentMethod === "cash" || paymentMethod === "credit" || paymentMethod === "debit") { console.log("Payment method accepted"); } // Default values with OR let name = null; let displayName = name || "Guest"; console.log(`Welcome, ${displayName}`); // Combining with AND let age = 17; let hasParentPermission = true; let isEmergency = false; if ((age >= 18 || hasParentPermission) && !isEmergency) { console.log("Access granted"); }

NOT Operator (!)

// NOT operator - inverts the condition let isLoggedIn = false; if (!isLoggedIn) { console.log("Please log in"); } // Double NOT for boolean conversion let value = "hello"; let isTruthy = !!value; console.log(isTruthy); // true // Practical examples let isMaintenanceMode = true; if (!isMaintenanceMode) { console.log("System is available"); } else { console.log("System is under maintenance"); } // Negating complex conditions let hasErrors = false; let isValidated = true; if (!hasErrors && isValidated) { console.log("Form submitted successfully"); } // Checking for empty arrays/objects let items = []; if (!items.length) { console.log("No items found"); }

Complex Logical Expressions

// Combining logical operators let user = { age: 25, isActive: true, role: "admin", hasPermission: true }; // Complex condition if ((user.age >= 18 && user.isActive) && (user.role === "admin" || user.hasPermission)) { console.log("Full access granted"); } // Using parentheses for clarity let temperature = 25; let isRaining = false; let hasUmbrella = true; if (temperature > 20 && (!isRaining || hasUmbrella)) { console.log("Good day to go outside"); } // Form validation example function validateForm(form) { let isValid = form.name && form.name.length > 0 && form.email && form.email.includes('@') && form.password && form.password.length >= 8; if (isValid) { console.log("Form is valid"); } else { console.log("Form has errors"); } } // Access control example function checkAccess(user, resource) { return (user.isAdmin || user.permissions.includes(resource)) && user.isActive && !user.isBanned; }

7. Advanced Conditional Patterns

Nullish Coalescing Operator (??)

// Nullish coalescing - returns right side if left is null or undefined let username = null; let displayName = username ?? "Guest"; console.log(displayName); // "Guest" // Difference from OR operator let count = 0; let defaultCount1 = count || 10; // 10 (0 is falsy) let defaultCount2 = count ?? 10; // 0 (0 is not nullish) // Practical example function getUserPreferences(user) { return { theme: user.theme ?? "light", language: user.language ?? "en", notifications: user.notifications ?? true }; } // Chaining nullish coalescing let config = { apiUrl: process.env.API_URL ?? config.defaultApiUrl ?? "http://localhost:3000" };

Optional Chaining (?.)

// Optional chaining - safely access nested properties let user = { name: "John", address: { street: "123 Main St", city: "New York" } }; // Safe property access let city = user.address?.city; let zipCode = user.address?.zipCode; // undefined, no error // Without optional chaining (old way) let cityOld = user.address && user.address.city; // Method chaining let result = user.getData?.(); let length = user.hobbies?.length; // Array access let firstHobby = user.hobbies?.[0]; let thirdHobby = user.hobbies?.[2]; // Practical example function displayUserInfo(user) { if (user?.profile?.isPublic) { console.log(`Name: ${user.name}`); console.log(`Email: ${user.contact?.email ?? "Not provided"}`); console.log(`City: ${user.address?.city ?? "Not provided"}`); } }

Guard Clauses

// Guard clauses - early returns to reduce nesting function processUser(user) { // Guard clauses if (!user) { console.log("No user provided"); return; } if (!user.isActive) { console.log("User is not active"); return; } if (!user.hasPermission) { console.log("User lacks permission"); return; } // Main logic console.log("Processing user:", user.name); // ... rest of the function } // Instead of nested if statements function processUserNested(user) { if (user) { if (user.isActive) { if (user.hasPermission) { console.log("Processing user:", user.name); // ... rest of the function } else { console.log("User lacks permission"); } } else { console.log("User is not active"); } } else { console.log("No user provided"); } } // Form validation with guard clauses function validateAndSubmitForm(formData) { if (!formData.name) { showError("Name is required"); return false; } if (!formData.email || !formData.email.includes('@')) { showError("Valid email is required"); return false; } if (!formData.password || formData.password.length < 8) { showError("Password must be at least 8 characters"); return false; } // If we get here, all validations passed submitForm(formData); return true; }

8. Common Patterns and Best Practices

‚úÖ Best Practices:

  • Use === and !== for strict equality comparisons
  • Keep conditions simple and readable
  • Use guard clauses to reduce nesting
  • Prefer early returns over deeply nested conditionals
  • Use meaningful variable names in conditions
  • Consider using switch for multiple value comparisons
  • Use logical operators to combine conditions efficiently
  • Always use braces {} even for single-line if statements

Common Validation Patterns

// String validation function validateString(str, minLength = 1) { if (typeof str !== 'string') { return false; } if (str.trim().length < minLength) { return false; } return true; } // Number validation function validateNumber(num, min=-Infinity, max=Infinity) { if (typeof num !=='number' || isNaN(num)) { return false; } if (num < min || num> max) { return false; } return true; } // Email validation function validateEmail(email) { if (!email || typeof email !== 'string') { return false; } const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/; return emailRegex.test(email); } // Array validation function validateArray(arr, minLength = 0) { if (!Array.isArray(arr)) { return false; } if (arr.length < minLength) { return false; } return true; } // Object validation function validateObject(obj, requiredKeys=[]) { if (!obj || typeof obj !=='object' || Array.isArray(obj)) { return false; } for (let key of requiredKeys) { if (!(key in obj)) { return false; } } return true; }

Error Handling Patterns

// Safe function execution function safeExecute(fn, fallback = null) { try { return fn(); } catch (error) { console.error('Function execution failed:', error); return fallback; } } // Conditional execution function executeIfValid(condition, action, errorMessage = "Condition not met") { if (condition) { return action(); } else { console.warn(errorMessage); return null; } } // Chain of responsibility pattern function processRequest(request) { if (handleAuth(request)) { return "Authentication handled"; } if (handleValidation(request)) { return "Validation handled"; } if (handleBusiness(request)) { return "Business logic handled"; } return "Request could not be processed"; } function handleAuth(request) { if (!request.token) { return false; } // Auth logic return true; } function handleValidation(request) { if (!request.data) { return false; } // Validation logic return true; } function handleBusiness(request) { // Business logic return true; }

‚ö†Ô∏è Common Pitfalls to Avoid:

  • Using == instead of === (type coercion issues)
  • Forgetting break statements in switch cases
  • Deep nesting of conditionals (pyramid of doom)
  • Not handling null/undefined values
  • Overly complex ternary operators
  • Not considering edge cases
  • Mixing different conditional styles inconsistently
