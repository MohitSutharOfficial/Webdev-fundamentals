🔍 Comparison Operators

Basic Comparisons

// Equality operators console.log(5 == 5); // true (loose equality) console.log(5 == "5"); // true (type conversion) console.log(5 === 5); // true (strict equality) console.log(5 === "5"); // false (no type conversion) // Inequality operators console.log(5 != 3); // true (loose inequality) console.log(5 != "5"); // false (type conversion) console.log(5 !== 3); // true (strict inequality) console.log(5 !== "5"); // true (no type conversion) // Relational operators console.log(10 > 5); // true console.log(10 >= 10); // true console.log(5 < 10); // true console.log(5 <=5); // true // String comparisons console.log("apple" < "banana" ); // true (lexicographic) console.log("10"> "9"); // false (string comparison) console.log("10" > 9); // true (number comparison) // Special values console.log(null == undefined); // true console.log(null === undefined); // false console.log(NaN == NaN); // false console.log(NaN === NaN); // false

Type Coercion in Comparisons

// Surprising results with type coercion console.log("" == 0); // true console.log("0" == 0); // true console.log(false == 0); // true console.log(null == 0); // false console.log(undefined == 0); // false // Arrays and objects console.log([1, 2] == "1,2"); // true console.log([1, 2] === "1,2"); // false console.log({} == {}); // false (different objects) console.log({} === {}); // false (different objects) // Best practices function safeEquals(a, b) { return a === b; } function safeCompare(a, b) { // Convert to same type first if (typeof a !== typeof b) { return false; } return a === b; } // Object comparison function deepEqual(obj1, obj2) { if (obj1 === obj2) return true; if (obj1 == null || obj2 == null) return false; if (typeof obj1 !== typeof obj2) return false; const keys1 = Object.keys(obj1); const keys2 = Object.keys(obj2); if (keys1.length !== keys2.length) return false; for (let key of keys1) { if (!keys2.includes(key)) return false; if (!deepEqual(obj1[key], obj2[key])) return false; } return true; }

🧠 Logical Operators

Boolean Logic

// AND operator (&&) console.log(true && true); // true console.log(true && false); // false console.log(false && true); // false console.log(false && false); // false // OR operator (||) console.log(true || true); // true console.log(true || false); // true console.log(false || true); // true console.log(false || false); // false // NOT operator (!) console.log(!true); // false console.log(!false); // true console.log(!!true); // true (double negation) // Practical examples let age = 25; let hasLicense = true; // Can drive if age >= 18 AND has license let canDrive = age >= 18 && hasLicense; // Can enter if age >= 21 OR has special permission let hasPermission = false; let canEnter = age >= 21 || hasPermission; // User is NOT admin let isAdmin = false; let isRegularUser = !isAdmin;

Short-Circuit Evaluation

"default"; // "default" let value2 = undefined ?? "default"; // "default" let value3 = 0 ?? "default"; // 0 let value4 = "" ?? "default"; // "" let value5 = false ?? "default"; // false // Logical assignment operators (ES2021) let x = 5; x &&= 10; // x = x && 10 (if x is truthy, set to 10) x ||= 20; // x = x || 20 (if x is falsy, set to 20) x ??= 30; // x = x ?? 30 (if x is nullish, set to 30)

Truthiness and Falsiness

// Falsy values console.log(Boolean(false)); // false console.log(Boolean(0)); // false console.log(Boolean(-0)); // false console.log(Boolean(0n)); // false (BigInt) console.log(Boolean("")); // false console.log(Boolean(null)); // false console.log(Boolean(undefined)); // false console.log(Boolean(NaN)); // false // Everything else is truthy console.log(Boolean(true)); // true console.log(Boolean(1)); // true console.log(Boolean(-1)); // true console.log(Boolean("hello")); // true console.log(Boolean(" ")); // true (space is truthy) console.log(Boolean([])); // true (empty array) console.log(Boolean({})); // true (empty object) console.log(Boolean(function(){})); // true // Practical use function isValidInput(input) { return input && input.trim().length > 0; } function hasValue(val) { return val !== null && val !== undefined && val !== ""; } // Converting to boolean let str = "hello"; let bool1 = Boolean(str); // true let bool2 = !!str; // true let bool3 = str ? true : false; // true

🔧 Assignment Operators

Basic Assignment

// Simple assignment let x = 5; let y = x; // y gets value of x // Multiple assignment let a, b, c; a = b = c = 10; // All get value 10 // Compound assignment operators let num = 10; num += 5; // num = num + 5 (15) num -= 3; // num = num - 3 (12) num *= 2; // num = num * 2 (24) num /= 4; // num = num / 4 (6) num %= 4; // num = num % 4 (2) num **= 3; // num = num ** 3 (8) // String concatenation assignment let greeting = "Hello"; greeting += " World"; // "Hello World" greeting += "!"; // "Hello World!" // Array assignment let arr = [1, 2, 3]; arr[0] = 10; // [10, 2, 3] arr[arr.length] = 4; // [10, 2, 3, 4] // Object assignment let person = {}; person.name = "John"; person["age"] = 30; person.city = "New York";

Destructuring Assignment

🔀 Bitwise Operators

Binary Operations

// Bitwise AND (&) console.log(5 & 3); // 1 (101 & 011 = 001) console.log(12 & 8); // 8 (1100 & 1000 = 1000) // Bitwise OR (|) console.log(5 | 3); // 7 (101 | 011 = 111) console.log(12 | 8); // 12 (1100 | 1000 = 1100) // Bitwise XOR (^) console.log(5 ^ 3); // 6 (101 ^ 011 = 110) console.log(12 ^ 8); // 4 (1100 ^ 1000 = 0100) // Bitwise NOT (~) console.log(~5); // -6 (inverts all bits) console.log(~0); // -1 // Left shift (<<) console.log(5 << 1); // 10 (101 << 1=1010) console.log(3 << 2); // 12 (11 << 2=1100) // Right shift (>>) console.log(10 >> 1); // 5 (1010 >> 1 = 101) console.log(12 >> 2); // 3 (1100 >> 2 = 11) // Unsigned right shift (>>>) console.log(-5 >>> 1); // 2147483645 (treats as unsigned) console.log(12 >>> 2); // 3 (same as >> for positive numbers)

Practical Applications

// Checking if number is even or odd function isEven(num) { return (num & 1) === 0; } console.log(isEven(4)); // true console.log(isEven(5)); // false // Multiplying/dividing by powers of 2 function multiplyBy2(num) { return num << 1; } function divideBy2(num) { return num>> 1; } console.log(multiplyBy2(5)); // 10 console.log(divideBy2(10)); // 5 // Swapping variables without temp function swap(a, b) { a = a ^ b; b = a ^ b; a = a ^ b; return [a, b]; } console.log(swap(5, 3)); // [3, 5] // Checking if power of 2 function isPowerOf2(num) { return num > 0 && (num & (num - 1)) === 0; } console.log(isPowerOf2(8)); // true console.log(isPowerOf2(6)); // false // Toggling bits (flags) let flags = 0; const FLAG_A = 1; // 001 const FLAG_B = 2; // 010 const FLAG_C = 4; // 100 // Set flags flags |= FLAG_A; // Set flag A flags |= FLAG_C; // Set flag C // Check flags console.log(flags & FLAG_A ? "A is set" : "A is not set"); console.log(flags & FLAG_B ? "B is set" : "B is not set"); // Clear flags flags &= ~FLAG_A; // Clear flag A // Toggle flags flags ^= FLAG_B; // Toggle flag B

🎯 Operator Best Practices

Comparison

  • Always use strict equality (===) over loose equality (==)
  • Use Object.is() for NaN and -0 comparisons
  • Be aware of type coercion in comparisons
  • Use explicit type conversion when needed
  • Avoid comparing different types without conversion
  • Use helper functions for complex comparisons

Logical

  • Use parentheses to clarify complex logical expressions
  • Leverage short-circuit evaluation for performance
  • Use nullish coalescing (??) for default values
  • Avoid deeply nested ternary operators
  • Use logical assignment operators when appropriate
  • Be explicit about truthy/falsy conversions

General

  • Use parentheses to make precedence clear
  • Avoid overly complex expressions
  • Use meaningful variable names
  • Comment complex operator combinations
  • Consider readability over brevity
  • Use linters to catch common mistakes

🚀 Modern Operator Features

Nullish Coalescing (??): Returns right operand when left is null or undefined.
Optional Chaining (?.): Safely access nested properties without errors.
Logical Assignment: Combine logical operators with assignment (&&=, ||=, ??=).
Pipeline Operator (|>): Proposal for functional programming style (future feature).
Exponentiation (**): Cleaner syntax for mathematical powers.
