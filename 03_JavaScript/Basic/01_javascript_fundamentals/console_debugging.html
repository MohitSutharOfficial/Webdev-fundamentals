JavaScript Console & Debugging

Master the art of debugging and console manipulation in JavaScript. Learn professional debugging techniques, console methods, and development tools.

1. Console Object Overview

The console object provides access to the browser's debugging console. It's an essential tool for development, testing, and debugging JavaScript applications.

Basic Console Methods

console.log()

General output and debugging

console.error()

Error messages with stack traces

console.warn()

Warning messages

console.info()

Informational messages

console.debug()

Debug-level messages

console.table()

Display data in tabular format

// Basic console methods console.log('Hello, World!'); console.error('This is an error message'); console.warn('This is a warning'); console.info('This is informational'); console.debug('This is debug information'); // Console with variables let name = 'JavaScript'; let version = 'ES2023'; console.log('Language:', name, 'Version:', version); // Multiple arguments console.log('User:', {name: 'John', age: 30}, 'Active:', true);

2. Advanced Console Methods

console.table() - Data Visualization

// Arrays const fruits = ['apple', 'banana', 'orange']; console.table(fruits); // Objects const user = { name: 'John Doe', age: 30, occupation: 'Developer' }; console.table(user); // Array of objects const users = [ {name: 'John', age: 30, role: 'Developer'}, {name: 'Jane', age: 25, role: 'Designer'}, {name: 'Bob', age: 35, role: 'Manager'} ]; console.table(users); // Specific columns console.table(users, ['name', 'role']);

console.group() - Organizing Output

// Grouping related logs console.group('User Information'); console.log('Name: John Doe'); console.log('Age: 30'); console.log('Role: Developer'); console.groupEnd(); // Nested groups console.group('Application Startup'); console.log('Loading configuration...'); console.group('Database Connection'); console.log('Connecting to database...'); console.log('Connection successful'); console.groupEnd(); console.log('Application ready'); console.groupEnd(); // Collapsed groups console.groupCollapsed('Debug Information'); console.log('Memory usage: 45MB'); console.log('CPU usage: 12%'); console.groupEnd();

console.time() - Performance Measurement

// Timing operations console.time('arrayLoop'); for (let i = 0; i < 1000000; i++) { // Some operation } console.timeEnd('arrayLoop'); // Multiple timers console.time('fetchData'); console.time('processData'); // Simulate async operations setTimeout(()=> { console.timeEnd('fetchData'); console.timeEnd('processData'); }, 1000); // Time logging console.time('calculation'); let sum = 0; for (let i = 0; i < 1000; i++) { sum +=i; } console.timeLog('calculation', 'Halfway done' ); // Continue calculation console.timeEnd('calculation');

3. Console Formatting and Styling

String Substitution

// String substitution patterns console.log('Hello %s', 'World'); console.log('Number: %d', 42); console.log('Float: %f', 3.14159); console.log('Object: %o', {name: 'John', age: 30}); console.log('DOM Element: %O', document.body); // Multiple substitutions let name = 'JavaScript'; let year = 2023; console.log('%s was created in %d', name, 1995); console.log('Current year: %d, Language: %s', year, name);

CSS Styling in Console

// Styled console output console.log('%cHello World!', 'color: red; font-size: 20px;'); console.log('%cSuccess!', 'color: green; font-weight: bold;'); console.log('%cError!', 'color: red; background: yellow; padding: 2px;'); // Multiple styles console.log( '%cBlue %cRed %cGreen', 'color: blue;', 'color: red;', 'color: green;' ); // Complex styling console.log( '%c JavaScript %c Console %c Styling', 'background: #222; color: #bada55; padding: 3px;', 'background: #f06; color: white; padding: 3px;', 'background: #0cf; color: #222; padding: 3px;' );

console.assert() - Conditional Logging

// Assertions let x = 5; console.assert(x > 0, 'x should be positive'); console.assert(x > 10, 'x should be greater than 10', x); // Function validation function divide(a, b) { console.assert(b !== 0, 'Division by zero!', {a, b}); return a / b; } divide(10, 2); // No assertion divide(10, 0); // Assertion fails // Array validation let arr = [1, 2, 3]; console.assert(Array.isArray(arr), 'Should be an array'); console.assert(arr.length > 0, 'Array should not be empty');

5. Debugging Techniques

Breakpoints and Debugging

// debugger statement function calculateTotal(items) { let total = 0; debugger; // Execution will pause here for (let item of items) { total += item.price; debugger; // Pause on each iteration } return total; } // Conditional breakpoints function processUser(user) { if (user.age < 18) { debugger; // Only pause for minors } return user; }

Error Handling and Logging

// Try-catch with console logging function riskyOperation() { try { console.log('Starting risky operation...'); // Potentially problematic code let result = someComplexCalculation(); console.log('Operation successful:', result); return result; } catch (error) { console.error('Operation failed:', error.message); console.error('Stack trace:', error.stack); console.group('Error Details'); console.log('Error name:', error.name); console.log('Error message:', error.message); console.log('Error stack:', error.stack); console.groupEnd(); throw error; } finally { console.log('Cleanup operations...'); } } // Custom error logging function logError(error, context = {}) { console.group(`❌ Error: ${error.message}`); console.error('Full error:', error); console.table(context); console.trace('Error trace'); console.groupEnd(); }

Performance Debugging

// Performance measurement function measurePerformance(fn, label) { console.time(label); console.group(`🔍 Performance: ${label}`); const start = performance.now(); const result = fn(); const end = performance.now(); console.log(`Execution time: ${end - start} milliseconds`); console.log('Result:', result); console.groupEnd(); console.timeEnd(label); return result; } // Memory usage monitoring function checkMemory() { if (performance.memory) { console.table({ 'Used JS Heap Size': `${Math.round(performance.memory.usedJSHeapSize / 1024 / 1024)} MB`, 'Total JS Heap Size': `${Math.round(performance.memory.totalJSHeapSize / 1024 / 1024)} MB`, 'JS Heap Size Limit': `${Math.round(performance.memory.jsHeapSizeLimit / 1024 / 1024)} MB` }); } } // Function call tracking function trackFunctionCalls(fn, name) { return function(...args) { console.count(`${name} called`); console.log(`${name} arguments:`, args); const result = fn.apply(this, args); console.log(`${name} result:`, result); return result; }; }

6. Advanced Debugging Patterns

Debugging Utility Functions

// Debug utility class class Debug { static enabled = true; static level = 'info'; // 'error', 'warn', 'info', 'debug' static log(message, level = 'info', ...args) { if (!this.enabled) return; const levels = ['error', 'warn', 'info', 'debug']; const currentLevelIndex = levels.indexOf(this.level); const messageLevelIndex = levels.indexOf(level); if (messageLevelIndex <= currentLevelIndex) { const timestamp=new Date().toISOString(); console[level](`[${timestamp}] [${level.toUpperCase()}]`, message, ...args); } } static error(message, ...args) { this.log(message, 'error' , ...args); } static warn(message, ...args) { this.log(message, 'warn' , ...args); } static info(message, ...args) { this.log(message, 'info' , ...args); } static debug(message, ...args) { this.log(message, 'debug' , ...args); } static group(label, fn) { console.group(label); try { return fn(); } finally { console.groupEnd(); } } } // Usage Debug.info('Application started'); Debug.warn('This is a warning'); Debug.error('An error occurred'); Debug.debug('Debug information');

Object Inspection

// Deep object inspection function inspectObject(obj, name = 'Object') { console.group(`🔍 Inspecting: ${name}`); console.log('Type:', typeof obj); console.log('Constructor:', obj.constructor?.name); console.log('Is Array:', Array.isArray(obj)); console.log('Keys:', Object.keys(obj)); console.log('Values:', Object.values(obj)); console.log('Entries:', Object.entries(obj)); console.group('Properties'); for (let key in obj) { if (obj.hasOwnProperty(key)) { console.log(`${key}:`, obj[key], `(${typeof obj[key]})`); } } console.groupEnd(); console.group('Methods'); Object.getOwnPropertyNames(obj).forEach(prop => { if (typeof obj[prop] === 'function') { console.log(`${prop}():`, obj[prop]); } }); console.groupEnd(); console.groupEnd(); } // Function parameter inspection function inspectFunction(fn) { console.group(`🔧 Function: ${fn.name}`); console.log('Function string:', fn.toString()); console.log('Length (parameters):', fn.length); console.log('Name:', fn.name); console.groupEnd(); }

Async Debugging

// Promise debugging function debugPromise(promise, label) { console.log(`🔄 Starting: ${label}`); return promise .then(result => { console.log(`✅ Success: ${label}`, result); return result; }) .catch(error => { console.error(`❌ Error: ${label}`, error); throw error; }); } // Async function debugging async function debugAsyncOperation() { try { console.time('async-operation'); console.log('Starting async operation...'); const result1 = await debugPromise( fetch('/api/data'), 'Fetch API data' ); const result2 = await debugPromise( processData(result1), 'Process data' ); console.log('Final result:', result2); console.timeEnd('async-operation'); return result2; } catch (error) { console.error('Async operation failed:', error); throw error; } }

7. Best Practices and Tips

✅ Best Practices:

  • Use descriptive labels for console.time() and console.group()
  • Remove console.log() statements before production
  • Use console.table() for arrays and objects
  • Leverage console.assert() for runtime validation
  • Use console.trace() to understand call stacks
  • Group related console output with console.group()

⚠️ Common Pitfalls:

  • Forgetting to remove debug code in production
  • Over-relying on console.log() instead of using debugger
  • Not checking if console exists in older browsers
  • Logging sensitive information to console
  • Not using appropriate console methods for different message types

Production-Safe Debugging

// Environment-aware debugging const DEBUG = process.env.NODE_ENV !== 'production'; function debugLog(...args) { if (DEBUG && typeof console !== 'undefined') { console.log(...args); } } // Conditional debugging class Logger { constructor(enabled = false) { this.enabled = enabled; } log(...args) { if (this.enabled) { console.log(...args); } } error(...args) { if (this.enabled) { console.error(...args); } } } const logger = new Logger(DEBUG); // Safe console usage function safeConsole(method, ...args) { if (typeof console !== 'undefined' && console[method]) { console[method](...args); } }

Performance Considerations

// Efficient debugging function efficientDebug(condition, message, ...args) { if (condition && DEBUG) { console.log(message, ...args); } } // Lazy evaluation for expensive operations function expensiveDebug(getMessage) { if (DEBUG) { console.log(getMessage()); } } // Usage expensiveDebug(() => `Complex calculation: ${doExpensiveCalculation()}`);
