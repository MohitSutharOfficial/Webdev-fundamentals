// Create Intersection Observer const observer = new IntersectionObserver((entries) => { entries.forEach(entry => { if (entry.isIntersecting) { // Element is visible entry.target.classList.add('visible'); // Optional: Stop observing once animated observer.unobserve(entry.target); } }); }, { threshold: 0.1, // Trigger when 10% visible rootMargin: '0px 0px -50px 0px' // Trigger 50px before entering viewport }); // Observe all elements with the class document.querySelectorAll('.observed-element').forEach(el => { observer.observe(el); }); // Advanced usage with multiple thresholds const advancedObserver = new IntersectionObserver((entries) => { entries.forEach(entry => { const ratio = entry.intersectionRatio; if (ratio > 0.25) { entry.target.style.opacity = ratio; } }); }, { threshold: [0, 0.25, 0.5, 0.75, 1.0] });

Monitor element size changes efficiently, essential for responsive components and layout management.

Resize Observer Code:

// Create Resize Observer const resizeObserver = new ResizeObserver(entries => { entries.forEach(entry => { const { width, height } = entry.contentRect; // Update display const display = entry.target.querySelector('#boxDimensions'); if (display) { display.textContent = `${Math.round(width)}x${Math.round(height)}`; } // Responsive behavior based on size if (width < 150) { entry.target.style.fontSize='12px' ; } else if (width < 250) { entry.target.style.fontSize='16px' ; } else { entry.target.style.fontSize='20px' ; } // Custom event for size changes entry.target.dispatchEvent(new CustomEvent('sizechange', { detail: { width, height } })); }); }); // Observe element const targetElement=document.getElementById('resizableBox'); resizeObserver.observe(targetElement); // Listen for custom size change events targetElement.addEventListener('sizechange', (event)=> { console.log('Size changed:', event.detail); });

üî¨ Mutation Observer API

Watch for changes in the DOM tree, including attribute changes, node additions/removals, and text content modifications.

Mutation Observer Implementation:

// Create Mutation Observer const mutationObserver = new MutationObserver(mutations => { mutations.forEach(mutation => { switch(mutation.type) { case 'childList': if (mutation.addedNodes.length) { console.log('Added nodes:', mutation.addedNodes); } if (mutation.removedNodes.length) { console.log('Removed nodes:', mutation.removedNodes); } break; case 'attributes': console.log(`Attribute ${mutation.attributeName} changed on:`, mutation.target); break; case 'characterData': console.log('Text content changed:', mutation.target.textContent); break; } }); }); // Configure observer const config = { childList: true, // Watch for child node changes attributes: true, // Watch for attribute changes characterData: true, // Watch for text content changes subtree: true, // Watch entire subtree attributeOldValue: true, // Include old attribute values characterDataOldValue: true // Include old text values }; // Start observing const targetNode = document.getElementById('mutationTarget'); mutationObserver.observe(targetNode, config); // Stop observing when needed // mutationObserver.disconnect();

// Observe different types of performance entries const performanceObserver = new PerformanceObserver((list) => { list.getEntries().forEach((entry) => { switch(entry.entryType) { case 'navigation': console.log('Navigation timing:', { domContentLoaded: entry.domContentLoadedEventEnd - entry.domContentLoadedEventStart, loadComplete: entry.loadEventEnd - entry.loadEventStart, totalTime: entry.loadEventEnd - entry.fetchStart }); break; case 'paint': console.log(`${entry.name}: ${entry.startTime}ms`); break; case 'measure': console.log(`Custom measure ${entry.name}: ${entry.duration}ms`); break; case 'resource': if (entry.duration > 100) { console.warn('Slow resource:', entry.name, `${entry.duration}ms`); } break; } }); }); // Observe specific entry types performanceObserver.observe({ entryTypes: ['navigation', 'paint', 'measure', 'resource'] }); // Create custom performance marks and measures performance.mark('task-start'); // ... do some work ... performance.mark('task-end'); performance.measure('task-duration', 'task-start', 'task-end'); // Get specific performance data const navigationEntries = performance.getEntriesByType('navigation'); const paintEntries = performance.getEntriesByType('paint'); const resourceEntries = performance.getEntriesByType('resource');

// Main thread - creating and using a Web Worker let worker; function startWorker() { if (typeof(Worker) !== "undefined") { if (typeof(worker) == "undefined") { // Create worker from inline script const workerScript = ` // Worker script self.onmessage = function(e) { const { type, data } = e.data; switch(type) { case 'fibonacci': const result = fibonacci(data); self.postMessage({ type: 'fibonacci-result', result: result, input: data }); break; case 'prime-check': const isPrime = checkPrime(data); self.postMessage({ type: 'prime-result', result: isPrime, input: data }); break; } }; function fibonacci(n) { if (n <= 1) return n; return fibonacci(n - 1) + fibonacci(n - 2); } function checkPrime(num) { if (num <=1) return false; for (let i=2; i <=Math.sqrt(num); i++) { if (num % i===0) return false; } return true; } // Send initial message self.postMessage({ type: 'ready' , message: 'Worker is ready!' }); `; const blob=new Blob([workerScript], { type: 'application/javascript' }); worker=new Worker(URL.createObjectURL(blob)); // Handle messages from worker worker.onmessage=function(e) { const output=document.getElementById('workerOutput'); output.textContent +=`\\n${JSON.stringify(e.data, null, 2)}`; }; worker.onerror=function(error) { console.error('Worker error:', error); }; } } else { alert("Sorry! No Web Worker support."); } } // Send data to worker function sendToWorker() { if (worker) { worker.postMessage({ type: 'fibonacci' , data: 35 }); worker.postMessage({ type: 'prime-check' , data: 982451653 }); } } // Terminate worker function stopWorker() { if (worker) { worker.terminate(); worker=undefined; } }
